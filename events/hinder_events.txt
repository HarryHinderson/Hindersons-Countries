hinder_events.4 = {
    type = country_events
    hidden = yes

    duration = 3

    trigger = {
        c:YUG ?= THIS
    }
    immediate = {
        if = {
            limit = {
                OR = {
                    was_formed_from = BUL 
                    AND = {
                        any_scope_state = {
                            state_region = { is_homeland = bulgarian }
                            count > 1
                        }
                    }
                    
                } 
            }
            s:STATE_NORTHERN_THRACE = {
                add_claim = ROOT
            }
            s:STATE_BULGARIA = {
                add_claim = ROOT
            }
            s:STATE_SKOPIA = {
                add_claim = ROOT
            }
            s:STATE_DOBRUDJA = {
                add_claim = ROOT
            }
            s:STATE_WESTERN_THRACE = {
                add_claim = ROOT
            }
        }
        else = {
            remove_primary_culture = cu:bulgarian
        }
    }
}
