capital_berlin = {
	is_shown = {
		exists = c:PRU
		c:PRU = THIS
		any_scope_state = {
			NOR = {
				state_region = s:STATE_BRANDENBURG
				is_capital = yes
			}
		}
	}
	possible = {
		any_scope_state = {
			state_region = s:STATE_BRANDENBURG
			is_capital = no
			is_incorporated = yes
			is_split_state = no
		}
	}
	when_taken = {
		trigger_event = {
			id = hinder_events.1
			popup = yes
		}
	}
	ai_chance = {
		value = 100
	}
}

greater_serbia = {
	is_shown = {
		exists = c:SER
		c:SER = this
		NOT = { has_modifier = greater_serbia_achieved }
	}

	possible = {
		has_technology_researched = nationalism
        any_scope_state = {
            state_region = { is_homeland = serb }
            count > 4
        }
        NOT = { has_modifier = greater_serbia_achieved }
	}
	when_taken = {
		add_modifier = {
            name = greater_serbia_achieved
        }
        if = {
        	limit = {
        		any_scope_state = {
        			state_region = s:STATE_SKOPIA
        		}
        	}
        	s:STATE_SKOPIA = {
        		add_homeland = cu:serb
        	}
        	
        }
	}
	ai_chance = {
		value = 100
	}
}

greater_serb_ambition = {
	is_shown = {
        exists = c:SER
        c:SER = THIS
        has_technology_researched = nationalism
        hidden_trigger = {
			NOT = { has_variable = greater_serb_ambition_taken }
		}
    }
    possible = {
		any_scope_state = {
            state_region = { is_homeland = serb }
            count > 1
        }
    }
    when_taken = {
    	set_variable = {
			name = greater_serb_ambition_taken
			value = yes
		}
    	trigger_event = {
			id = hinder_events.2
			popup = yes
		}
    }
}
